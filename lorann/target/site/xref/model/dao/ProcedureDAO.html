48  	        final ArrayList<Example> procedure = new ArrayList<Example>();
49  	        final CallableStatement callStatement = prepareCall(sqlMapByLevel);
50  	        callStatement.setInt(1, Level);
51  	        if (callStatement.execute()) {
52  	            final ResultSet result = callStatement.getResultSet();
53  
54  	            for (boolean isResultLeft = result.first(); isResultLeft; isResultLeft = result.next()) {
55  	                procedure.add(new Example(result.getString(ElementIndex),result.getInt(XPositionIndex), result.getInt(YPositionIndex)));
56  	            }
57  	            result.close();
58  	        }
59  	        return procedure;
